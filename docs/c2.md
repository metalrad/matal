void swap(double *wk1,double *wk2)
{
	//double wk1,wk2;
	double w;
	w=*wk1; *wk1=*wk2; *wk2=w;
}
int sweep(double a[N][N], double b[N], double x[N])
	{
		int i,j,k,ik;
		double ak,aik,ww;
		int sw;
		
		for(k=0;k<N;k++)
		{
			ak=a[k][k];

		if(fabs(ak)<=EPSILON)
		{
		ik=k+1;
		while((ik<N)&&(fabs(a[ik][k])<EPSILON)) ik++;
		if(ik<N)
		{
			for(j=k;j<N;j++) {swap(&a[k][j],&a[ik][j]);}
			swap(&b[k],&b[ik]);
			ak=a[k][k];
		}
		else
		{
			printf("ピボットが0です\n");
			sw=FALSE;
			goto owari;
		}
	}
	for(j=k;j<N;j++) a[k][j]=a[k][j]/ak;
	b[k]=b[k]/ak;
	for(i=0;i<N;i++)
	{
	if(i!=k)
	{
		aik=a[i][k];
		for(j=k;j<N;j++) a[i][j]=a[i][j]-aik*a[k][j];
		b[i]=b[i]-b[k]*aik;
	}
}
		
}

for(k=0;k<N;k++) 
	x[k]=b[k];
owari:;

return (int)ak;

}
